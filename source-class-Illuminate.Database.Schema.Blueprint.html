 69:      *
 70:      * @param  \Illuminate\Database\Connection  $connection

 72:      * @return array
 73:      */
 74:     public function toSql(Connection $connection, Grammar $grammar)
 75:     {
 76:         $this->addImpliedCommands();
 77: 
 78:         $statements = array();
 79: 
 80:         // Each type of command has a corresponding compiler function on the schema
 81:         // grammar which is used to build the necessary SQL statements to build
 82:         // the blueprint element, so we'll just call that compilers function.
 83:         foreach ($this->commands as $command)
 84:         {
 85:             $method = 'compile'.ucfirst($command->name);
 86: 
 87:             if (method_exists($grammar, $method))
 88:             {
 89:                 if ( ! is_null($sql = $grammar->$method($this, $command, $connection)))
 90:                 {
 91:                     $statements = array_merge($statements, (array) $sql);
 92:                 }
 93:             }
 94:         }
 95:

 97:     }

120:     {

122:         {
123:             foreach (array('primary', 'unique', 'index') as $index)
124:             {
125:                 // If the index has been specified on the given column, but is simply
126:                 // equal to "true" (boolean), no name has been specified for this
127:                 // index, so we will simply call the index methods without one.
128:                 if ($column->$index === true)
129:                 {
130:                     $this->$index($column->name);
131: 
132:                     continue 2;
133:                 }
134: 
135:                 // If the index has been specified on the column and it is something
136:                 // other than boolean true, we will assume a name was provided on
137:                 // the index specification, and pass in the name to the method.
138:                 elseif (isset($column->$index))
139:                 {
140:                     $this->$index($column->name, $column->$index);
141: 
142:                     continue 2;
143:                 }
144:             }
145:         }

693:     protected function dropIndexCommand($command, $type, $index)
694:     {
695:         $columns = array();
696: 
697:         // If the given "index" is actually an array of columns, the developer means
